\def\Shade{\pgfmathsetmacro\I{int(\i)}\ifnum\I>140 \tikzset{@/.style={right color=\InnerColor,left color=\OuterColor}}
\def\Offshift{0}
\ifnum\I>180 \tikzset{@/.style={top color=\InnerColor,bottom color=\OuterColor}}
   \ifnum\I<220 \tikzset{@/.style={right color=\InnerColor,left color=\OuterColor}}
   \else\ifnum\I>320\tikzset{@/.style={left color=\InnerColor,right color=\OuterColor}}
   \fi
\fi\fi
\else\ifnum\I<50 \tikzset{@/.style={left color=\InnerColor,right color=\OuterColor}}
\else \tikzset{@/.style={bottom color=\InnerColor,top color=\OuterColor}}
\fi\fi}

\def\DrawLeaf#1#2;{\scope[rotate around={\i-90:(0,0)}]
\Shade

\endscope}

% #1 is length | #2 inner | #3 outer | #4 step
\def\DrawRound#1#2#3#4{%
\gdef\Length{#1}%
\def\InnerColor{#2!96!\OuterColor}%
\def\OuterColor{#3}%
% \def\Init{0}%
\def\Step{#4}%
\pgfmathsetmacro\Steptarget{int(2*\Step)}%
\pgfmathsetmacro\Stepgoal{int(360-2*\Step-1)}%
\foreach \i in {\Step,\Steptarget,...,\Stepgoal} {%
\DrawLeaf\fill;
}
\scope
\pgfinterruptboundingbox
\edef\i{\Step}{\def\Shade{}\tikzset{@default/.style={}}\DrawLeaf{\path[save path=\clippath,pingu@invclip]};}%
\endpgfinterruptboundingbox
\clip[use path=\clippath];
\pgfmathsetmacro\Cur{int(360-2*\Step)}
\edef\i{\Cur}\DrawLeaf\draw;
\pgfmathsetmacro\Cur{int(360-\Step)}
\edef\i{\Cur}\DrawLeaf\draw;
\def\i{0}\DrawLeaf\draw;
\endscope
\scope
\pgfinterruptboundingbox
\pgfmathsetmacro\Cur{int(2*\Step)}
\def\i{\Cur}{\def\Shade{}\tikzset{@default/.style={}}\DrawLeaf{\path[save path=\clippath,pingu@invclip]};}
\endpgfinterruptboundingbox
\scope[rotate around={\Step-90:(0,0)}]
\clip[use path=\clippath];
\draw[very thin] (0,0) to[bend right=4] ++(5.75pt,\Length) arc(-9:189:5.75pt and 6pt);
\endscope
\endscope
}

\def\Bloom#1{%
\scope[transparency group, opacity=.55] % 625
    \ifnum\Circles>5 \scope[rotate around={5:(0,0)}]
    \DrawRound{2}{@inner}{#1!50!black}{8}
    \endscope\fi
    \ifnum\Circles>4 \DrawRound{1.75}{@inner}{#1!50!black}{10}\fi
    \ifnum\Circles>3 \scope[rotate around={5:(0,0)}]
    \DrawRound{1.5}{@inner}{#1!70!black}{12}
    \endscope\fi
    \ifnum\Circles>2 \DrawRound{1.25}{@inner}{#1!80!black}{15}\fi
    \ifnum\Circles>1 \scope[rotate around={-8:(0,0)}]
    \scope[transparency group,opacity=.385]\DrawRound{1}{@inner}{#1}{22.5}\endscope
    \endscope\fi
    \DrawRound{.75}{@inner!98!black}{#1}{36}
\endscope
    \fill[outer color=black!98!@inner,inner color=black!90!@inner] (0,0) circle[radius=\CenterSize];
}
